1. 🌟 Overview

The DebugChannel system provides a flexible, categorized logging framework for C++ applications. It allows developers to create named debug channels that can output to both console and log files simultaneously, with configurable output destinations and formatting.

Key Features:
  • ✅ Named debug channels for categorized logging
  • ✅ Dual output support (console and file logging)
  • ✅ Configurable output flags and destinations
  • ✅ Printf-style formatted output
  • ✅ Warning and assertion support
  • ✅ Automatic log file management
  • ✅ Centralized channel management
  • ✅ Thread-safe singleton pattern

9. 🚀 Advanced Usage

Conditional Logging

// Conditional logging patterns class GameSystem { private: DebugChannels::CDebugChannel m_debug; bool m_verboseLogging; public: GameSystem(const std::string& name, bool verbose = false) : m_debug(name), m_verboseLogging(verbose) { if (!verbose) { m_debug.SetOutputFlags(DebugChannels::OF_Log); // File only for non-verbose } } void LogInfo(const char* format, ...) { if (m_verboseLogging) { va_list args; va_start(args, format); // Manual formatting for custom handling char buffer[1024]; vsnprintf(buffer, sizeof(buffer), format, args); m_debug.print(std::string(buffer)); va_end(args); } } void LogError(const char* format, ...) { // Always log errors regardless of verbose setting va_list args; va_start(args, format); char buffer[1024]; vsnprintf(buffer, sizeof(buffer), format, args); m_debug.warning("ERROR: %s", buffer); // Use warning for emphasis va_end(args); } };

Performance-Critical Logging

// Macro-based conditional compilation for performance #ifdef DEBUG_BUILD #define PERF_LOG(channel, ...) channel.printf(__VA_ARGS__) #define PERF_ASSERT(channel, condition, ...) channel.Assert(condition, __VA_ARGS__) #else #define PERF_LOG(channel, ...) ((void)0) #define PERF_ASSERT(channel, condition, ...) ((void)0) #endif // Usage in performance-critical code class RenderSystem { DebugChannels::CDebugChannel m_debug{"Rendering"}; public: void RenderFrame() { PERF_LOG(m_debug, "Starting frame render\n"); // Performance critical rendering code for (auto& object : renderObjects) { PERF_ASSERT(m_debug, object.IsValid(), "Invalid render object at index %d\n", i); RenderObject(object); } PERF_LOG(m_debug, "Frame render completed\n"); } };

Channel Scoping and RAII

// RAII wrapper for automatic scope logging class ScopedDebugLog { private: DebugChannels::CDebugChannel& m_channel; std::string m_scopeName; std::chrono::high_resolution_clock::time_point m_startTime; public: ScopedDebugLog(DebugChannels::CDebugChannel& channel, const std::string& scopeName) : m_channel(channel), m_scopeName(scopeName) { m_startTime = std::chrono::high_resolution_clock::now(); m_channel.printf("ENTER: %s\n", m_scopeName.c_str()); } ~ScopedDebugLog() { auto endTime = std::chrono::high_resolution_clock::now(); auto duration = std::chrono::duration_cast<std::chrono::microseconds>( endTime - m_startTime); m_channel.printf("EXIT: %s (took %ld μs)\n", m_scopeName.c_str(), duration.count()); } }; // Usage void ComplexFunction() { ScopedDebugLog scope(CoreDebug, "ComplexFunction"); // Function implementation // Automatic entry/exit logging with timing }

10. 📊 Logging Patterns

Structured Logging

// Structured logging with consistent formatting class StructuredLogger { private: DebugChannels::CDebugChannel& m_channel; public: StructuredLogger(DebugChannels::CDebugChannel& channel) : m_channel(channel) {} void LogEvent(const std::string& category, const std::string& event, const std::string& details = "") { auto now = std::chrono::system_clock::now(); auto time_t = std::chrono::system_clock::to_time_t(now); m_channel.printf("[%s][%s] %s", std::ctime(&time_t), category.c_str(), event.c_str()); if (!details.empty()) { m_channel.printf(" - %s", details.c_str()); } m_channel.print("\n"); } void LogMetric(const std::string& name, double value, const std::string& unit = "") { m_channel.printf("[METRIC] %s: %.3f %s\n", name.c_str(), value, unit.c_str()); } void LogError(const std::string& error, const std::string& context = "") { m_channel.printf("[ERROR] %s", error.c_str()); if (!context.empty()) { m_channel.printf(" (Context: %s)", context.c_str()); } m_channel.print("\n"); } }; // Usage StructuredLogger logger(GameplayDebug); logger.LogEvent("PLAYER", "Level Up", "Level 5 reached"); logger.LogMetric("FPS", 60.0, "frames/sec"); logger.LogError("Save failed", "Disk full");

Error Severity Levels

// Custom severity levels using different methods class SeverityLogger { private: DebugChannels::CDebugChannel& m_channel; public: SeverityLogger(DebugChannels::CDebugChannel& channel) : m_channel(channel) {} void Info(const char* format, ...) { va_list args; va_start(args, format); char buffer[1024]; vsnprintf(buffer, sizeof(buffer), format, args); m_channel.printf("[INFO] %s\n", buffer); va_end(args); } void Warning(const char* format, ...) { va_list args; va_start(args, format); char buffer[1024]; vsnprintf(buffer, sizeof(buffer), format, args); m_channel.warning("%s\n", buffer); // Uses built-in warning format va_end(args); } void Error(const char* format, ...) { va_list args; va_start(args, format); char buffer[1024]; vsnprintf(buffer, sizeof(buffer), format, args); m_channel.printf("[ERROR] %s\n", buffer); va_end(args); } void Critical(const char* format, ...) { va_list args; va_start(args, format); char buffer[1024]; vsnprintf(buffer, sizeof(buffer), format, args); // Critical errors always assert false for immediate attention m_channel.Assert(false, "[CRITICAL] %s\n", buffer); va_end(args); } };

12. ✅ Best Practices

🎯 Logging Guidelines

  • Use descriptive channel names - Make channels easily identifiable
  • Include context in messages - Add relevant parameters and state
  • Use appropriate severity levels - Reserve assertions for critical failures
  • Format consistently - Establish logging patterns across systems
  • Consider performance impact - Use conditional compilation for heavy logging

✅ Good Practices

// ✅ Good: Descriptive messages with context GameplayDebug.printf("Player %d took %d damage from %s (health: %d/%d)\n", playerId, damage, source.c_str(), currentHealth, maxHealth); // ✅ Good: Meaningful channel names DebugChannels::CDebugChannel NetworkProtocol("NetworkProtocol"); DebugChannels::CDebugChannel AssetLoader("AssetLoader"); // ✅ Good: Conditional verbose logging #ifdef VERBOSE_LOGGING RenderDebug.printf("Rendering object %d at (%.2f, %.2f, %.2f)\n", objId, pos.x, pos.y, pos.z); #endif // ✅ Good: Appropriate use of assertions bool success = InitializeSystem(); SystemDebug.Assert(success, "Failed to initialize critical system!\n");

⚠️ Common Pitfalls

// ❌ Poor: Vague messages without context GameplayDebug.print("Error occurred\n"); // ❌ Poor: Generic channel names DebugChannels::CDebugChannel Debug1("Debug"); DebugChannels::CDebugChannel Debug2("System"); // ❌ Poor: Excessive logging in tight loops for (int i = 0; i < 10000; ++i) { ProcessDebug.printf("Processing item %d\n", i); // Performance killer! } // ❌ Poor: Using assertions for non-critical conditions NetworkDebug.Assert(connectionEstablished, "Connection not ready\n"); // Too harsh

Performance Considerations

// Performance-conscious logging patterns class PerformanceLogger { DebugChannels::CDebugChannel& m_channel; bool m_enabled; public: PerformanceLogger(DebugChannels::CDebugChannel& channel) : m_channel(channel), m_enabled(true) {} void SetEnabled(bool enabled) { m_enabled = enabled; } template<typename... Args> void Log(const char* format, Args&&... args) { if (m_enabled) { m_channel.printf(format, std::forward<Args>(args)...); } } // Batch logging for high-frequency events void LogBatch(const std::vector<std::string>& messages) { if (m_enabled && !messages.empty()) { std::string batch; for (const auto& msg : messages) { batch += msg + "\n"; } m_channel.print(batch); } } };

13. 🔧 Troubleshooting

Common Issues and Solutions

Issue Symptoms Solution
No console output Messages only in log files Check if OF_StdOut flag is set
No log files created Messages only on console Verify OF_Log flag and file permissions
Log files empty Files created but no content Ensure proper log flushing and file closure
Assertions not working No assertion dialogs Check if running in debug mode
Performance issues Slow execution with logging Use conditional compilation for heavy logging
