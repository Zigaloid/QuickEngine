1. 🌟 Overview

The FileSystem API provides a comprehensive, cross-platform file system abstraction layer for C++20 applications. It offers type-safe error handling, flexible filesystem stacking, and consistent behavior across different platforms and storage backends.

Key Features:
  • ✅ Cross-platform file system abstraction
  • ✅ Type-safe error handling with Result<T>
  • ✅ Stackable filesystem implementations
  • ✅ RAII-based resource management
  • ✅ Comprehensive file and directory operations
  • ✅ Path manipulation utilities
  • ✅ Support for special directories (temp, user, app)
  • ✅ Pattern-based file/directory filtering

8. 🔧 FileSystemManager

The FileSystemManager implements a composite pattern, allowing multiple filesystem implementations to be stacked with fallback behavior.

Manager Operations

// Stack management void PushFileSystem(std::unique_ptr<IFileSystem> fileSystem); // Add to front (highest priority) void AddFileSystem(std::unique_ptr<IFileSystem> fileSystem); // Add to back (lowest priority) std::unique_ptr<IFileSystem> PopFileSystem(); // Remove from front std::unique_ptr<IFileSystem> RemoveFileSystem(); // Remove from back void ClearFileSystems(); // Remove all // Information size_t GetFileSystemCount() const; bool IsEmpty() const;

Stacking Example

// Create filesystem manager FileSystemManager manager; // Add filesystem implementations in priority order manager.PushFileSystem(std::make_unique<ArchiveFileSystem>("game.pak")); // Highest priority manager.AddFileSystem(std::make_unique<NetworkFileSystem>("cdn.server.com")); // Medium priority manager.AddFileSystem(std::make_unique<StandardFileSystem>()); // Fallback // Operations try each filesystem in order auto result = manager.ReadAllText("config.txt"); // 1. Try ArchiveFileSystem first // 2. If fails, try NetworkFileSystem // 3. If fails, try StandardFileSystem // 4. If all fail, return error

Custom File Processing

// Process large files in chunks class FileProcessor { public: static Result<void> ProcessLargeFile(IFileSystem* fs, const std::string& inputPath, const std::string& outputPath) { auto inputResult = fs->OpenFile(inputPath, FileMode::Read); if (inputResult.HasError()) { return Result<void>("Failed to open input: " + inputResult.GetError()); } auto outputResult = fs->CreateFile(outputPath); if (outputResult.HasError()) { return Result<void>("Failed to create output: " + outputResult.GetError()); } auto inputFile = std::move(inputResult.GetValue()); auto outputFile = std::move(outputResult.GetValue()); const size_t bufferSize = 8192; std::vector<uint8_t> buffer(bufferSize); while (true) { auto readResult = inputFile->Read(buffer.data(), bufferSize); if (readResult.HasError()) { return Result<void>("Read error: " + readResult.GetError()); } size_t bytesRead = readResult.GetValue(); if (bytesRead == 0) break; // End of file // Process data (example: convert to uppercase) for (size_t i = 0; i < bytesRead; ++i) { if (buffer[i] >= 'a' && buffer[i] <= 'z') { buffer[i] = buffer[i] - 'a' + 'A'; } } auto writeResult = outputFile->Write(buffer.data(), bytesRead); if (writeResult.HasError()) { return Result<void>("Write error: " + writeResult.GetError()); } } outputFile->Flush(); return Result<void>(); } };

Directory Tree Walking

// Recursive directory traversal class DirectoryWalker { public: static void WalkDirectory(IFileSystem* fs, const std::string& path, std::function<void(const FileInfo&)> fileCallback, std::function<void(const FileInfo&)> dirCallback = nullptr) { auto dirResult = fs->OpenDirectory(path); if (dirResult.HasError()) { std::cerr << "Failed to open directory: " << dirResult.GetError() << std::endl; return; } auto directory = std::move(dirResult.GetValue()); // Process all entries auto entriesResult = directory->GetEntries(); if (entriesResult.HasError()) { std::cerr << "Failed to get entries: " << entriesResult.GetError() << std::endl; return; } for (const auto& entry : entriesResult.GetValue()) { if (entry.isDirectory) { if (dirCallback) { dirCallback(entry); } // Recursive call WalkDirectory(fs, entry.fullPath, fileCallback, dirCallback); } else { fileCallback(entry); } } } // Find all files with specific extension static std::vector<std::string> FindFilesByExtension(IFileSystem* fs, const std::string& rootPath, const std::string& extension) { std::vector<std::string> foundFiles; WalkDirectory(fs, rootPath, [&](const FileInfo& fileInfo) { if (fs->GetExtension(fileInfo.fullPath) == extension) { foundFiles.push_back(fileInfo.fullPath); } }); return foundFiles; } }; // Usage auto* fileSystem = Core::CoreSystem::GetFileSystemManager(); // Find all .cpp files recursively auto cppFiles = DirectoryWalker::FindFilesByExtension(fileSystem, "src", ".cpp"); std::cout << "Found " << cppFiles.size() << " C++ files:" << std::endl; for (const auto& file : cppFiles) { std::cout << " " << file << std::endl; }

// Configuration management system class ConfigManager { private: IFileSystem* fileSystem_; std::string configPath_; std::map<std::string, std::string> settings_; public: ConfigManager(IFileSystem* fs, const std::string& configFile) : fileSystem_(fs), configPath_(configFile) { LoadSettings(); } void LoadSettings() { if (!fileSystem_->Exists(configPath_)) { std::cout << "Config file not found, using defaults" << std::endl; return; } auto textResult = fileSystem_->ReadAllText(configPath_); if (textResult.HasError()) { std::cerr << "Failed to read config: " << textResult.GetError() << std::endl; return; } // Parse simple key=value format std::istringstream stream(textResult.GetValue()); std::string line; while (std::getline(stream, line)) { size_t pos = line.find('='); if (pos != std::string::npos) { std::string key = line.substr(0, pos); std::string value = line.substr(pos + 1); settings_[key] = value; } } } void SaveSettings() { std::stringstream content; for (const auto& [key, value] : settings_) { content << key << "=" << value << "\n"; } auto writeResult = fileSystem_->WriteAllText(configPath_, content.str()); if (writeResult.HasError()) { std::cerr << "Failed to save config: " << writeResult.GetError() << std::endl; } } std::string GetSetting(const std::string& key, const std::string& defaultValue = "") { auto it = settings_.find(key); return (it != settings_.end()) ? it->second : defaultValue; } void SetSetting(const std::string& key, const std::string& value) { settings_[key] = value; } void CreateBackup() { if (fileSystem_->Exists(configPath_)) { std::string backupPath = configPath_ + ".backup"; auto result = fileSystem_->CopyFile(configPath_, backupPath, true); if (result.IsSuccess()) { std::cout << "Config backup created: " << backupPath << std::endl; } } } };

13. ✅ Best Practices

🎯 Performance Guidelines

  • Use RAII patterns - Files automatically close when Result objects are destroyed
  • Batch operations - Read/write in reasonable chunk sizes (4KB-64KB)
  • Check file existence - Use Exists() before expensive operations
  • Prefer utility methods - Use ReadAllText/WriteAllText for simple operations
  • Handle large files carefully - Stream data instead of loading everything into memory

✅ Good Practices

// ✅ Good: Always check results auto result = fileSystem->ReadAllText("config.txt"); if (result.IsSuccess()) { ProcessConfiguration(result.GetValue()); } else { HandleConfigError(result.GetError()); } // ✅ Good: Use appropriate access modes auto file = fileSystem->OpenFile("data.bin", FileMode::ReadWrite); // ✅ Good: Validate paths before operations if (fileSystem->Exists(inputPath) && fileSystem->IsFile(inputPath)) { // Safe to proceed } // ✅ Good: Use path utilities std::string fullPath = fileSystem->CombinePath(baseDir, fileName); // ✅ Good: Handle special directories properly auto tempResult = fileSystem->GetTempDirectory(); if (tempResult.IsSuccess()) { std::string tempFile = fileSystem->CombinePath(tempResult.GetValue(), "temp.dat"); }

⚠️ Common Pitfalls

// ❌ Poor: Ignoring error results fileSystem->WriteAllText("file.txt", content); // What if this fails? // ❌ Poor: Not checking file existence auto file = fileSystem->OpenFile("nonexistent.txt", FileMode::Read); // This will fail! // ❌ Poor: Hardcoded path separators std::string path = "dir\\file.txt"; // Won't work on Unix systems // ❌ Poor: Loading large files into memory auto content = fileSystem->ReadAllText("huge_file.dat"); // Memory issues! // ❌ Poor: Not handling special characters in paths auto result = fileSystem->CreateFile("file with spaces.txt");
